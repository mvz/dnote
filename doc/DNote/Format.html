def render
  if notes.empty?
    $stderr << "No #{notes.labels.join(', ')} notes.\n"
  else
    case format
    when 'custom'
      render_custom
    else
      render_template
    end
  end
end

def render_template
  template = File.join(File.dirname(__FILE__), 'templates', "#{format}.erb")
  raise "No such format - #{format}" unless File.exist?(template)
  result = erb(template)
  publish(result)  
end
